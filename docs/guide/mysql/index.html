                # 获取随机数

                公式：SELECT FLOOR(最小数+RAND()*(最大数–最小数+1))

                #获取5-10号之间的随机数
SELECT LPAD(FLOOR(5+RAND()*(10-5+1)),2,'0')

#获取8-20点之间的随机数
SELECT LPAD(FLOOR(8+RAND()*(20-8+1)),2,'0')

#获取0-60分之间的随机数
SELECT LPAD(FLOOR(0+RAND()*(60-0+1)),2,'0')
SELECT LPAD(FLOOR(RAND()*61),2,'0')
