            # 3. 使用代理发送请求（json格式数据）

            String paramJson = "{\"key1\":\""+value1+"\",\"key2\":\""+value2+"\",\"key3\":"+value3+"}";
String result2 = HttpRequest.post(url).setHttpProxy("10.12.2.22",8080)
                    .header("header1", value1)//头信息，多个头信息多次调用此方法即可

                    .header("header3", value3)//头信息，多个头信息多次调用此方法即可
