                            # Linux 使用指南

                            # 1. 切换工作目录：cd 命令

                            # 1.1. 进入到根目录

                            cd /
                            1

                            # 1.2. 返回进入此目录之前所在目录

                            cd -
                            1

                            # 1.3. 返回上一级目录

                            cd ..
                            1

                            # 1.4. 返回上两级目录（或更多级目录）

                            cd ../..
                            1

                            # 2. 查看当前路径：pwd 命令

                            pwd
                            1

                            # 3. 切换 root 账号

                            su - root    # ---> 回车，输入 root 账号的密码
                            1

                            # 4. 改名或移动：mv 命令

                            命令用来为文件或目录改名、或将文件或目录移入其它位置。

                            #改名：将aa（目录或文件）改名为bb
mv aa bb
#移动：将aa（目录或文件）移动到cc目录下
mv aa cc/
#移动并改名：将aa（目录或文件）移动到cc目录下，并改名为bb
mv aa cc/bb
